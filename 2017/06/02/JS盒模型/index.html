JS盒模型

JS盒子模型

JS中提供了与盒子模型相关的属性
CSS盒子模型属性：width ， height ， border ， margin ， padding
13个属性中，只有与scrollLeft和scrollTop可以设置值，其他只能获取值

client系列(跟溢出的内容无关)
  • clientWidth = width+padding(左右)
  • clientHeight = height+padding(上下)
  • clientLeft 左边框
  • clientTop 上边框
    offset系列(跟溢出内容无关)
    • offsetWidth = width + padding(左右) + border(左右) = clientWidth + border(左右)
    • offsetHeight = height + padding(上下) + border(上下) = clientHeight + border(左右)
    • 与偏移量相关
  • offsetParent 参照物
  • offsetLeft 左偏移量
  • offsetTop 上偏移量
    scroll系列 (跟溢出内容有关)
  • scrollWidth ≈ width（获取真实的宽度）+ 左padding值 约等于真实宽度
  • scrollHeight ≈ height（获取真实的高度）+ 上padding值 约等于真实的高度
    为什么是约等于？
  • 各个浏览器的行高不一样
  • 相同浏览器是否设置overflow属性，值也不一样
  • 跟滚动条相关
    • scrollLeft 横向卷出去的内容
    • scrollTop 纵向卷出去的内容

（二）盒子偏移量和OFFET

offsetParent：当前盒子的父级参照物

offsetTop / offsetLeft：获取当前盒子距离其父级参照物的偏移量（上偏移 / 左偏移）从当前盒子的外边框开始~父级参照物的内边框

参照物：最近的，已经定位的父级元素，若没有找到已经定位的父级元素，则它的参照物是body

偏移量：当前元素外边框到参照物内边框的距离

（1）同一平面中，元素的父级参照物和结构没有必然联系，默认她们的父级参照物都是body（当前平面最外层的盒子）
body的父级参照物是null

（2）参照物可以改变：构建不同的平面即可（使用z-index，但是这个属性只对定位有作用），所以改变元素的定位（position : relative / absolute / fixed）可以改变其父级参照物

第一种方式：offset：获取当前元素距离body的偏移（左偏移和上偏移）

let offset=function(curEle){

    //1.先获取当前元素本身的左/上偏移
    let curLeft=curEle.offsetLeft,
    curTop=curEle.offsetTop;
    p=curEle.offsetParent;

    //2.累加父级参照物的边框和偏移（一直向上找，找到body为止，每当找到一个父级参照物都把它的边框和偏移累加起来，根据元素不一样，具体找几次也不知道（用while循环））
    //tagName获取当前元素的标签名（大写的）
    while(p.tagName!=='BODY'){//当找到的父级参照物是BODY结束查找和累加操作
    //3.把找到的父级参照物的边框和偏移值累加起来
    curLeft+=p.clientLeft;
    curLeft+=p.offsetLeft;
    curTop+=p.clientTop;
    curTop+=p.offsetTop;
    p=p.offsetParent;//基于当前找到的父级参照物继续向上查找
    }
}

第二种方式：不管当前元素的参照物是谁，要求当前元素到body的偏移量

 var inner=document.getElementById('inner');
 //不管当前元素的参照物是谁，要求当前元素到body的偏移量
 function offset(ele){
    //先获取当前元素到参照物的偏移量
    var l=ele.offsetLeft;
    var t=ele.offsetTop;
    var p=ele.offsetParent;
    //判断参照物是不是body,如果不是继续累加，反之，结束
    while(p&&p!==document.body){//p判断要排除ele是body元素的情况，和排除p是body的情况。
        l+= p.offsetLeft+ p.clientLeft;
        t+= p.offsetTop+ p.clientTop;
        p= p.offsetParent;
    }
    return {
        l:l,
        t:t
    }
}
console.log(offset(inner).l);//传入的参数ele，只能是对象

let css=function(...arg){
    //...arg是剩余运算符，传递的实参集合。因为ES6不支持arguments，基本上arguments相当于是...arg

（四）封装，实现set-Group-css批量设置元素的样式

for in ：循环，遍历一个对象中的键值对的，有多少组键值对，我们就遍历多少次

setGroupCss：给元素批量设置样式

let setGroupCss=function(curEle,options={}){
    //遍历传递的options，有多少键值对就循环多少次，每一次都调取set-css方法逐一设置即可
    for(let attr in options){
        if(!options.hasOwnProperty(attr)) break;
        //options：传递进来的需要修改的样式对象（集合）
        //attr：每一次遍历到的集合中的某一项（要操作的样式属性名）
        //options[attr]：传递的要操作的样式属性值
        setCss(curEle,attr,options[attr]);
    }
};
setGroupCss(outer,{
    width:400,
    height:400,
    padding:30
});
