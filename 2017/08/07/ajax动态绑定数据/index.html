var xhr = new XMLHttpRequest();
xhr.open("get","./data.json",false);
var data = null;
xhr.onreadystatechange = function(){
    if(xhr.readyState==4 && /^2\d{2}$/.test(xhr.status)){
        data = JSON.parse(xhr.responseText);
    }
}
xhr.send(null);
